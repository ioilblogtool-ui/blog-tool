import { getCollection } from "astro:content";
import BaseLayout from "../layouts/BaseLayout.astro";
import SiteHeader from "../components/SiteHeader.astro";

const normalizedBase = import.meta.env.BASE_URL.endsWith("/")
  ? import.meta.env.BASE_URL
  : `${import.meta.env.BASE_URL}/`;
const withBase = (path) => `${normalizedBase}${path.replace(/^\//, "")}`;
const toToolPath = (tool) => tool.data.slug ?? tool.id.replace(/\.md$/, "");
const tools = (await getCollection("tools")).sort((a, b) => a.data.order - b.data.order);
const featured = tools.slice(0, 4);

// 전체 카운트
const toolCount = tools.length;

// 카테고리 unique 추출
const categories = [...new Set(tools.map((t) => t.data.eyebrow).filter(Boolean))];

// 리포트 컬렉션 (없으면 빈 배열)
let reports = [];
try {
  reports = await getCollection("reports");
} catch {
  reports = [];
}
const reportCount = reports.length || 2;

// featured: data.featured === true 인 것 우선, 없으면 slice(0,4)
const featuredTools = tools.filter((t) => t.data.featured).slice(0, 4);
const showcaseTools = featuredTools.length >= 2 ? featuredTools : tools.slice(0, 4);

// title/description 수정

### ④ 전체 계산기 섹션 (신규)
섹션 헤더:

  오른쪽: <a href={withBase("/tools/")}>전체 보기 →</a>
  class: home-section__head

  "전체" + categories 배열
  class: home-filter__row
  각 chip: class="home-filter__chip" data-category="전체" 또는 카테고리명
  첫 번째(전체)는 class에 is-active 추가

tools 전체 2열 그리드 리스트 (class: home-tools-list):
  각 행 (class: home-tool-row):
    data-category={tool.data.eyebrow}
    href={withBase(`/tools/${toToolPath(tool)}/`)}
    왼쪽: 제목(13px 500) + 카테고리(10px secondary) 세로 배치
    오른쪽: "→" 화살표

필터 JS (인라인 script):
  .home-filter__chip 클릭 시:
    - 클릭된 chip에 is-active 클래스
    - 나머지 chip에서 is-active 제거
    - "전체" 선택 시 모든 .home-tool-row 표시
    - 특정 카테고리 선택 시 data-category 매칭되는 행만 표시
    - 나머지는 display: none
